<script>
const FIXAS=['Aluguel','Financiamento','Água','Luz','Energia solar','Internet','Telefone celular','Plano de saúde','Seguro veicular','OAB Alcides','OAB Natalia','Streaming e assinaturas','Academia','Funcionários / folha de pagamento','Impostos','Dízimo / doações','Cartão de crédito'];
const VARIAVEIS=['Mercado','Alimentação fora de casa','Delivery','Combustível','Transporte','Farmácia','Saúde','Educação','Lazer','Vestuário','Manutenção do veículo','Taxas bancárias','Emergência','Reserva'];
const PALETTE=['#378ADD','#1D9E75','#BA7517','#D4537E','#7F77DD','#D85A30','#639922','#888780','#185FA5','#0F6E56','#854F0B','#993556','#534AB7','#993C1D','#3B6D11','#5F5E5A','#0C447C','#085041','#633806','#72243E'];
const MONTHS=['Janeiro','Fevereiro','Março','Abril','Maio','Junho','Julho','Agosto','Setembro','Outubro','Novembro','Dezembro'];

let expenses=[],budgets={},history=[],colorMap={},colorIdx=0,editingId=null;
let donutChart,barChart,pizzaChart,compareChart,histChart;
let currentDate=new Date();

function getColor(cat){if(!colorMap[cat])colorMap[cat]=PALETTE[colorIdx++%PALETTE.length];return colorMap[cat];}
function fmt(v){return v.toLocaleString('pt-BR',{style:'currency',currency:'BRL'});}
function monthKey(d){return`${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}`;}
function monthLabel(d){return`${MONTHS[d.getMonth()]} ${d.getFullYear()}`;}

function changeMonth(dir){
  currentDate=new Date(currentDate.getFullYear(),currentDate.getMonth()+dir,1);
  document.getElementById('month-label').textContent=monthLabel(currentDate);
}

function showPage(p){
  document.querySelectorAll('.page').forEach(el=>el.classList.remove('active'));
  document.querySelectorAll('.tab').forEach(el=>el.classList.remove('active'));
  document.getElementById('page-'+p).classList.add('active');
  const idx={despesas:0,orcamento:1,historico:2}[p];
  document.querySelectorAll('.tab')[idx].classList.add('active');
  if(p==='orcamento')renderOrcamento();
  if(p==='historico')renderHistorico();
}

function updateCatOptions(){
  const cats=document.getElementById('tipo').value==='Fixa'?FIXAS:VARIAVEIS;
  document.getElementById('cat').innerHTML=cats.map(c=>`<option value="${c}">${c}</option>`).join('');
}
function updateOrcCatOptions(){
  const cats=document.getElementById('orc-tipo').value==='Fixa'?FIXAS:VARIAVEIS;
  document.getElementById('orc-cat').innerHTML=cats.map(c=>`<option value="${c}">${c}</option>`).join('');
}
function updateEditCatOptions(){
  const cats=document.getElementById('e-tipo').value==='Fixa'?FIXAS:VARIAVEIS;
  document.getElementById('e-cat').innerHTML=cats.map(c=>`<option value="${c}">${c}</option>`).join('');
}

function addExpense(){
  const desc=document.getElementById('desc').value.trim();
  const tipo=document.getElementById('tipo').value;
  const cat=document.getElementById('cat').value;
  const val=parseFloat(document.getElementById('val').value);
  if(isNaN(val)||val<=0){document.getElementById('val').focus();return;}
  expenses.push({id:Date.now(),desc,tipo,cat,val});
  document.getElementById('desc').value='';
  document.getElementById('val').value='';
  renderAll();
}

function removeExpense(id){expenses=expenses.filter(e=>e.id!==id);renderAll();}

function openEdit(id){
  const e=expenses.find(x=>x.id===id);if(!e)return;
  editingId=id;
  document.getElementById('e-desc').value=e.desc;
  document.getElementById('e-tipo').value=e.tipo;
  updateEditCatOptions();
  document.getElementById('e-cat').value=e.cat;
  document.getElementById('e-val').value=e.val;
  document.getElementById('modal').classList.add('open');
}
function closeModal(){document.getElementById('modal').classList.remove('open');editingId=null;}
function saveEdit(){
  const idx=expenses.findIndex(x=>x.id===editingId);if(idx<0)return;
  const val=parseFloat(document.getElementById('e-val').value);
  if(isNaN(val)||val<=0)return;
  expenses[idx]={...expenses[idx],desc:document.getElementById('e-desc').value.trim(),tipo:document.getElementById('e-tipo').value,cat:document.getElementById('e-cat').value,val};
  closeModal();renderAll();
}

function clearData(){
  if(!confirm('Limpar todos os lançamentos do mês atual?'))return;
  expenses=[];renderAll();
}

function exportCSV(){
  if(!expenses.length){alert('Nenhum dado para exportar.');return;}
  const rows=[['Descrição','Tipo','Categoria','Valor'],...expenses.map(e=>[e.desc||'',e.tipo,e.cat,e.val.toFixed(2).replace('.',',')])];
  const csv=rows.map(r=>r.map(v=>`"${v}"`).join(';')).join('\n');
  const blob=new Blob(['\uFEFF'+csv],{type:'text/csv;charset=utf-8'});
  const a=document.createElement('a');a.href=URL.createObjectURL(blob);
  a.download=`despesas_${monthKey(currentDate)}.csv`;a.click();
}

function saveBudget(){
  const cat=document.getElementById('orc-cat').value;
  const val=parseFloat(document.getElementById('orc-val').value);
  if(isNaN(val)||val<=0){document.getElementById('orc-val').focus();return;}
  budgets[cat]=val;document.getElementById('orc-val').value='';renderOrcamento();
}
function removeBudget(cat){delete budgets[cat];renderOrcamento();}

function saveCurrentMonth(){
  const total=expenses.reduce((s,e)=>s+e.val,0);
  if(!expenses.length){alert('Nenhuma despesa para salvar.');return;}
  const key=monthKey(currentDate);
  const existing=history.findIndex(h=>h.key===key);
  const entry={key,label:monthLabel(currentDate),total,count:expenses.length,cats:getCatTotals(),expenses:[...expenses]};
  if(existing>=0)history[existing]=entry;
  else history.push(entry);
  history.sort((a,b)=>a.key.localeCompare(b.key));
  renderHistorico();
  alert(`${monthLabel(currentDate)} salvo no histórico!`);
}

function getCatTotals(){
  const m={};expenses.forEach(e=>{m[e.cat]=(m[e.cat]||0)+e.val;});return m;
}

function renderAll(){renderTable();renderMetrics();renderDespesasCharts();renderRendaSaldo();}

function renderRendaSaldo(){
  const renda=parseFloat(document.getElementById('renda').value)||0;
  const total=expenses.reduce((s,e)=>s+e.val,0);
  const saldo=renda-total;
  const el=document.getElementById('saldo-renda');
  const elPct=document.getElementById('pct-renda');
  if(!renda){el.textContent='—';elPct.textContent='';return;}
  el.textContent=fmt(saldo);
  el.style.color=saldo>=0?'#3B6D11':'#A32D2D';
  const pct=Math.round(total/renda*100);
  elPct.textContent=`(${pct}% da renda comprometido)`;
}

function renderTable(){
  const tbody=document.getElementById('tbody');
  if(!expenses.length){tbody.innerHTML='<tr><td colspan="5" class="empty">Nenhuma despesa. Adicione a primeira acima.</td></tr>';return;}
  tbody.innerHTML=[...expenses].reverse().map(e=>{
    const cls=e.tipo==='Fixa'?'badge-fixa':'badge-variavel';
    const c=getColor(e.cat);
    return`<tr>
      <td><span style="display:inline-flex;align-items:center;gap:5px"><span style="width:7px;height:7px;border-radius:2px;background:${c};flex-shrink:0"></span>${e.cat}</span></td>
      <td style="color:var(--color-text-secondary)">${e.desc||'—'}</td>
      <td><span class="badge ${cls}">${e.tipo}</span></td>
      <td style="text-align:right;font-weight:500">${fmt(e.val)}</td>
      <td style="text-align:center">
        <button class="btn-edit" onclick="openEdit(${e.id})" aria-label="Editar"><i class="ti ti-pencil"></i></button>

}

function renderOrcamento(){
  const catTotals=getCatTotals();
  const totalOrc=Object.values(budgets).reduce((s,v)=>s+v,0);
  const totalGasto=expenses.reduce((s,e)=>s+e.val,0);
  const saldo=totalOrc-totalGasto;
  const pct=totalOrc>0?Math.round(totalGasto/totalOrc*100):0;
  document.getElementById('o-total').textContent=fmt(totalOrc);
  document.getElementById('o-gasto').textContent=fmt(totalGasto);
  document.getElementById('o-saldo').textContent=fmt(saldo);
  document.getElementById('o-saldo').style.color=saldo>=0?'#3B6D11':'#A32D2D';
  document.getElementById('o-pct').textContent=pct+'%';
  document.getElementById('o-pct').style.color=pct>100?'#A32D2D':pct>80?'#854F0B':'#3B6D11';
  let html='';
  if(totalOrc>0){
    if(totalGasto>totalOrc) html+=`<div class="alert-box alert-danger"><i class="ti ti-alert-triangle" style="font-size:17px;flex-shrink:0"></i><div><strong>Orçamento ultrapassado!</strong> Você gastou ${fmt(totalGasto-totalOrc)} a mais que o planejado.</div></div>`;
    else if(pct>=80) html+=`<div class="alert-box alert-warning"><i class="ti ti-alert-circle" style="font-size:17px;flex-shrink:0"></i><div><strong>Atenção:</strong> ${pct}% do orçamento utilizado. Saldo restante: ${fmt(saldo)}.</div></div>`;
    else html+=`<div class="alert-box alert-success"><i class="ti ti-circle-check" style="font-size:17px;flex-shrink:0"></i><div>Orçamento sob controle. Saldo disponível: <strong>${fmt(saldo)}</strong> (${100-pct}% restante).</div></div>`;
  }
  Object.entries(budgets).forEach(([cat,lim])=>{
    const g=catTotals[cat]||0;
    if(g>lim) html+=`<div class="alert-box alert-danger"><i class="ti ti-alert-triangle" style="font-size:15px;flex-shrink:0"></i><div><strong>${cat}</strong>: ultrapassou em ${fmt(g-lim)}.</div></div>`;
    else if(lim>0&&g/lim>=0.8) html+=`<div class="alert-box alert-warning"><i class="ti ti-alert-circle" style="font-size:15px;flex-shrink:0"></i><div><strong>${cat}</strong>: ${Math.round(g/lim*100)}% do limite usado.</div></div>`;
  });
  document.getElementById('alertas').innerHTML=html;
  if(pizzaChart)pizzaChart.destroy();
  if(compareChart)compareChart.destroy();
  const allCats=Object.keys(catTotals);
  if(allCats.length){
    const pLabels=allCats,pValues=allCats.map(c=>catTotals[c]),pColors=allCats.map(c=>getColor(c));
    const totalG=pValues.reduce((s,v)=>s+v,0);
    document.getElementById('legend-pizza').innerHTML=pLabels.map((l,i)=>{const p=totalG>0?Math.round(pValues[i]/totalG*100):0;return`<span class="legend-item"><span class="legend-dot" style="background:${pColors[i]}"></span>${l} ${p}%</span>`;}).join('');
    pizzaChart=new Chart(document.getElementById('chartPizza'),{type:'pie',data:{labels:pLabels,datasets:[{data:pValues,backgroundColor:pColors,borderWidth:2,borderColor:'transparent'}]},options:{responsive:true,maintainAspectRatio:false,plugins:{legend:{display:false},tooltip:{callbacks:{label:ctx=>{const p=totalG>0?Math.round(ctx.raw/totalG*100):0;return` ${fmt(ctx.raw)} (${p}%)`;}}}}}});
  } else document.getElementById('legend-pizza').innerHTML='';
  const budCats=Object.keys(budgets);
  if(budCats.length){
    const cH=Math.max(budCats.length*40+60,80);
    document.getElementById('compareWrap').style.height=cH+'px';
    compareChart=new Chart(document.getElementById('chartCompare'),{type:'bar',data:{labels:budCats,datasets:[{label:'Orçado',data:budCats.map(c=>budgets[c]),backgroundColor:'rgba(55,138,221,0.2)',borderColor:'#378ADD',borderWidth:1.5,borderRadius:3,borderSkipped:false},{label:'Realizado',data:budCats.map(c=>catTotals[c]||0),backgroundColor:budCats.map(c=>{const g=catTotals[c]||0,l=budgets[c];return g>l?'rgba(163,45,45,.7)':g/l>=.8?'rgba(186,117,23,.7)':'rgba(29,158,117,.7)';}),borderRadius:3,borderSkipped:false}]},options:{indexAxis:'y',responsive:true,maintainAspectRatio:false,plugins:{legend:{display:true,position:'bottom',labels:{font:{size:11},boxWidth:9}},tooltip:{callbacks:{label:ctx=>` ${fmt(ctx.raw)}`}}},scales:{x:{ticks:{callback:v=>'R$'+(v>=1000?(v/1000).toFixed(0)+'k':v),font:{size:10}},grid:{color:'rgba(128,128,128,0.08)'}},y:{ticks:{font:{size:11}},grid:{display:false}}}}});
  }
  const orcList=document.getElementById('orc-list');
  if(!Object.keys(budgets).length){orcList.innerHTML='<p class="empty">Nenhum limite definido.</p>';return;}
  orcList.innerHTML=Object.entries(budgets).map(([cat,lim])=>{
    const g=catTotals[cat]||0,p=lim>0?Math.round(g/lim*100):0,over=g>lim,warn=!over&&p>=80;
    const barColor=over?'#E24B4A':warn?'#EF9F27':'#1D9E75',pctColor=over?'#A32D2D':warn?'#854F0B':'#3B6D11';
    return`<div class="orcamento-item">
      <div class="orc-header">
        <span class="orc-cat"><span style="width:7px;height:7px;border-radius:2px;background:${getColor(cat)};flex-shrink:0"></span>${cat}</span>
        <div style="display:flex;align-items:center;gap:10px">
          <span class="orc-vals">${fmt(g)} / ${fmt(lim)}</span>
          <span class="orc-pct" style="color:${pctColor}">${p}%</span>
          <button class="btn-del" onclick="removeBudget('${cat}')" aria-label="Remover"><i class="ti ti-x"></i></button>
        </div>
      </div>
      <div class="progress-bar-bg"><div class="progress-bar-fill" style="width:${Math.min(p,100)}%;background:${barColor}"></div></div>
      ${over?`<p style="font-size:11px;color:#A32D2D;margin-top:3px"><i class="ti ti-alert-triangle"></i> Ultrapassou em ${fmt(g-lim)}</p>`:''}
    </div>`;
  }).join('');
}

function renderHistorico(){
  if(histChart)histChart.destroy();
  const count=history.length;
  const acum=history.reduce((s,h)=>s+h.total,0);
  const media=count?acum/count:0;
  const maior=count?history.reduce((a,b)=>a.total>b.total?a:b):null;
  document.getElementById('h-count').textContent=count;
  document.getElementById('h-acum').textContent=fmt(acum);
  document.getElementById('h-media').textContent=fmt(media);
  document.getElementById('h-maior').textContent=maior?`${maior.label} (${fmt(maior.total)})`:'—';
  if(count){
    histChart=new Chart(document.getElementById('chartHist'),{
      type:'line',
      data:{labels:history.map(h=>h.label.substring(0,3)+' '+String(h.key).substring(0,4)),datasets:[{label:'Total gasto',data:history.map(h=>h.total),borderColor:'#378ADD',backgroundColor:'rgba(55,138,221,0.08)',borderWidth:2,pointRadius:4,pointBackgroundColor:'#378ADD',fill:true,tension:.3}]},
      options:{responsive:true,maintainAspectRatio:false,plugins:{legend:{display:false},tooltip:{callbacks:{label:ctx=>` ${fmt(ctx.raw)}`}}},scales:{x:{ticks:{font:{size:11}},grid:{display:false}},y:{ticks:{callback:v=>'R$'+(v>=1000?(v/1000).toFixed(0)+'k':v),font:{size:10}},grid:{color:'rgba(128,128,128,0.08)'}}}}
    });
  }
  const list=document.getElementById('hist-list');
  if(!count){list.innerHTML='<p class="empty">Nenhum mês salvo ainda.</p>';return;}
  list.innerHTML=[...history].reverse().map((h,i)=>`
    <div class="hist-month" onclick="toggleHistDetail('hd-${i}')">
      <div><div class="hist-name">${h.label}</div><div class="hist-meta">${h.count} lançamentos</div></div>
      <div style="display:flex;align-items:center;gap:10px">
        <span class="hist-val">${fmt(h.total)}</span>
        <i class="ti ti-chevron-down" style="font-size:14px;color:var(--color-text-secondary)"></i>
      </div>
    </div>
    <div class="hist-detail" id="hd-${i}">
      ${Object.entries(h.cats).sort((a,b)=>b[1]-a[1]).map(([cat,val])=>`<div class="hist-row"><span>${cat}</span><span style="font-weight:500">${fmt(val)}</span></div>`).join('')}
    </div>
  `).join('');
}
